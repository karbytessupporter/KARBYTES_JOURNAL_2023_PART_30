Yes, it is possible to define your own log and exp functions.

The following code defines basic log and exp functions for the purpose of the program:

#include< iostream >
#include< iomanip >
#include < cmath >

using namespace std;

double log(double x){
  double e = 2.71828;
  double log_x = 0; 
  while(x > e){
    log_x++;
    x /= e;
  }
  return log_x;
}
 
double exp(double x){ 
    double e = 2.71828; 
    double exp_x = 1; 
    while(x > 0){ 
        exp_x *= e; 
        x--; 
    } 
    return exp_x;
}
